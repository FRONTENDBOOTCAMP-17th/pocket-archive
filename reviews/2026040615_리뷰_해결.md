## 3-1. 게시글 상세 - DOM이 없을 때 initPostDetail 호출 (중요도: 높음)

### 문제

`src/main.js` 86~97번째 줄에서 `initPostDetail(2)`를 호출한 **후에** HTML을 `innerHTML`에 넣고 있어서, `postDetailContent` 요소를 찾지 못해요.

### 해결 방법

**Step 1.** `src/main.js`의 `loadPage()` 함수 안에서 순서를 변경합니다.

```js
// 수정 전 (src/main.js 79~97번째 줄)
if (current === 'board') {
  import('./components/board.js');
}

if (page.includes("pokedex.html")) {
  initPokedex();
}
if (page.includes("detailPost.html")) {
  console.log(`2번 게시글 상세페이지 로드 중...`);
  initPostDetail(2);
}

document.getElementById('content').innerHTML = html;

if (page.includes("myparty.html")) {
  const { init } = await import("./scripts/myparty.js");
  init();
}

// 수정 후 — HTML을 먼저 넣고, 그 다음에 초기화!
document.getElementById('content').innerHTML = html;

if (current === 'board' && !page.includes("detailPost.html")) {
  await import('./components/board.js');
}
if (page.includes("pokedex.html")) {
  initPokedex();
}
if (page.includes("detailPost.html")) {
  initPostDetail(postId);  // 3-2번도 함께 해결!
}
if (page.includes("myparty.html")) {
  const { init } = await import("./scripts/myparty.js");
  init();
}
```

핵심 원칙: **"HTML 삽입 → 초기화 함수 호출"** 순서를 지키세요.

> **쉽게 말하면:** 빈 방에 가구를 배치하려면, 먼저 방이 있어야 하잖아요. HTML(방)을 먼저 만들고, 그 다음에 JavaScript(가구 배치)를 실행해야 해요.

## 3-3. mypage의 my.js — 초기화 함수로 감싸기 (중요도: 높음)

### 문제

`src/components/mypage/my.js`가 import되는 즉시 DOM에 접근하고 이벤트를 등록해요. HTML이 아직 로드되지 않았으면 에러가 발생해요.

### 해결 방법

**Step 1.** `src/components/mypage/my.js` 전체를 초기화 함수로 감싸기

```js
// 수정 전 (src/components/mypage/my.js 전체)
import { initMyInfo } from './myinfo.js';
import { initMyBoard } from './myboard.js';

const btnMyInfo = document.getElementById('btn-myinfo');
const btnMyBoard = document.getElementById('btn-myboard');
const btnLogout = document.getElementById('btn-logout');
const btnWithdraw = document.getElementById('btn-withdraw');

function setActiveTab(activeBtn) { ... }

btnMyInfo.addEventListener('click', () => { ... });
btnMyBoard.addEventListener('click', () => { ... });
btnLogout.addEventListener('click', () => { ... });
btnWithdraw.addEventListener('click', () => { ... });

initMyInfo();

// 수정 후 — export 함수로 감싸기
import { initMyInfo } from './myinfo.js';
import { initMyBoard } from './myboard.js';

export function initMyPage() {
  const btnMyInfo = document.getElementById('btn-myinfo');
  const btnMyBoard = document.getElementById('btn-myboard');
  const btnLogout = document.getElementById('btn-logout');
  const btnWithdraw = document.getElementById('btn-withdraw');

  if (!btnMyInfo || !btnMyBoard) return;  // 방어 코드 추가

  function setActiveTab(activeBtn) {
    [btnMyInfo, btnMyBoard].forEach((btn) => {
      const isActive = btn === activeBtn;
      btn.classList.toggle('bg-[#e6f7f5]', isActive);
      btn.classList.toggle('text-[#00bba7]', isActive);
      btn.classList.toggle('border', !isActive);
      btn.classList.toggle('border-[#00bba7]/25', !isActive);
      btn.classList.toggle('text-[#4a7a72]', !isActive);
      btn.classList.toggle('max-[1025px]:bg-[#22A9DA]/40!', isActive);
      btn.classList.toggle('max-[1025px]:bg-[#F3F4F6]!', !isActive);
    });
  }

  btnMyInfo.addEventListener('click', () => {
    setActiveTab(btnMyInfo);
    initMyInfo();
    window.scrollTo({ top: 0, behavior: 'smooth' });
  });

  btnMyBoard.addEventListener('click', () => {
    setActiveTab(btnMyBoard);
    initMyBoard();
    window.scrollTo({ top: 0, behavior: 'smooth' });
  });

  btnLogout.addEventListener('click', () => {
    if (confirm('로그아웃 하시겠습니까?')) {
      window.location.href = '/';
    }
  });

  btnWithdraw.addEventListener('click', () => {
    if (confirm('정말 탈퇴하시겠습니까? 모든 데이터가 삭제됩니다.')) {
      alert('회원탈퇴가 완료되었습니다.');
      window.location.href = '/';
    }
  });

  // 기본 탭: 내 정보
  initMyInfo();
}
```

**Step 2.** `src/main.js`에서 mypage 로딩 부분 수정

```js
// src/main.js — loadPage() 함수 안에 추가
if (current === "mypage") {
  document.getElementById('content').innerHTML = html;
  const { initMyPage } = await import('./components/mypage/my.js');
  initMyPage();
}
```

> **쉽게 말하면:** 손님(HTML)이 가게에 도착하기 전에 주문(이벤트 등록)을 받으면 안 돼요. 손님이 도착한 다음에 "뭐 드릴까요?" 하고 물어봐야 해요.

## 3-7. 에러 메시지 innerHTML → textContent (중요도: 중간)

### 문제

`login.js`와 `register.js`에서 에러 메시지를 `innerHTML`로 설정하고 있어요.

### 해결 방법

**login.js** (119, 121, 125, 129, 131, 133번째 줄):

```js
// 수정 전
msg.innerHTML = '아이디를 입력해주세요';
msg.innerHTML = '';
msg.innerHTML = '비밀번호를 입력해주세요';

// 수정 후
msg.textContent = '아이디를 입력해주세요';
```

**register.js** (184, 188, 192, 197, 201, 205, 210, 214, 218, 223번째 줄):

모든 `msg.innerHTML =`을 `msg.textContent =`으로 변경하세요.

> **쉽게 말하면:** `innerHTML`은 편지 내용을 실행하는 로봇이에요. `textContent`는 내용을 읽기만 하는 안전한 로봇이에요.

## 3-8. boardDetailUI.js XSS 방지 (중요도: 중간)

### 문제

`src/components/board/boardDetailUI.js`에서 사용자 데이터를 `innerHTML`에 직접 삽입하고 있어요.

### 해결 방법

**Step 1.** 공통 유틸리티 파일 생성

```js
// src/utils/escapeHtml.js (새 파일)
export function escapeHtml(text) {
  const div = document.createElement("div");
  div.textContent = text;
  return div.innerHTML;
}
```

**Step 2.** `boardDetailUI.js`에서 사용

```js
// 수정 후 (src/components/board/boardDetailUI.js)
import { escapeHtml } from '../../utils/escapeHtml.js';

export const Comment = (comment) => `
  <div class="bg-[#F8F9FA] rounded-2xl border border-gray-100" 
       style="padding: 20px 24px; margin-bottom: 5px;">
    <div class="flex justify-between items-center mb-2">
      <span class="font-bold text-gray-800 text-[15px]">${escapeHtml(comment.authorNickname)}</span>
      <span class="text-xs text-gray-400 font-medium">
        ${comment.createdAt ? escapeHtml(comment.createdAt.split("T")[0].replace(/-/g, ".")) : ""}
      </span>
    </div>
    <p class="text-gray-600 text-[14px] text-left leading-relaxed">${escapeHtml(comment.content)}</p>
  </div>
`;
```

`BoardDetailContent`도 동일하게 `${post.title}`, `${post.author}`, `${post.category}`, `${post.content}` 부분을 `${escapeHtml(...)}`으로 감싸세요.

**Step 3.** `myparty.js`의 `escapeHtml` 도 공통 유틸리티로 교체

```js
// src/scripts/myparty.js

// 기존 escapeHtml 함수 정의 삭제 (438~442번째 줄)
```

## 3-16. window 전역 함수 제거 (이벤트 위임으로 교체) (중요도: 낮음)

### 문제

`pokedexUI.js`에서 `window.toggleSidebar`, `pokedex.js`에서 `window.selectPokemon`을 사용하고 있어요.

### 해결 방법

이 항목은 수정 범위가 넓어서 시간이 있을 때 도전해보세요.

**Step 1.** `pokedexUI.js`에서 `onclick` 속성 대신 `data-no` 속성 사용

```js
// 수정 전 (pokedexUI.js 24~30번째 줄)
export const SidebarItem = (p) => `
  <div onclick="selectPokemon(${p.no})" ...>
    No.${p.no} ${p.name}
  </div>
`;

// 수정 후 — onclick 제거, data-no 추가
export const SidebarItem = (p) => `
  <div data-no="${p.no}" style="cursor: pointer;" 
       class="p-3 text-sm text-gray-400 rounded-xl hover:bg-[#E8F5E9] hover:text-[#05B29F] transition-all">
    No.${p.no} ${p.name}
  </div>
`;
```

**Step 2.** `pokedex.js`에서 이벤트 위임으로 처리

```js
// 수정 후 (pokedex.js — renderSidebar 함수 뒤에 추가)
function setupSidebarClick() {
  const list = document.getElementById("sidebarList");
  if (!list) return;
  list.addEventListener("click", (e) => {
    const item = e.target.closest("[data-no]");
    if (item) selectPokemon(Number(item.dataset.no));
  });
}

// initPokedex()에서 setupSidebarClick() 호출
export async function initPokedex() {
  const res = await fetch("/pokemon_full_ko.json");
  allPokemon = await res.json();
  filteredPokemon = [...allPokemon];
  
  setupSearch();
  renderSidebar();
  setupSidebarClick();  // 이벤트 위임 등록 (한 번만!)
  renderGrid(1);
}
```

**Step 3.** `window.selectPokemon`을 일반 함수로 변경

```js
// 수정 전
window.selectPokemon = async function (no) { ... };

// 수정 후
async function selectPokemon(no) { ... }
```

> **쉽게 말하면:** 전역 함수는 "누구나 건드릴 수 있는 공용 리모컨"이에요. 자기만 쓰는 리모컨(함수)을 만들면 다른 사람이 실수로 건드릴 일이 없어요.
